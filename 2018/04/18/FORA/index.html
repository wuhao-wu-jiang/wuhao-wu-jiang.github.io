\begin{align} & r(s, v_i) / k_i - b \le 0 &\forall i \in [1, n] \\\\ & -k_i \le 0 &\forall i \in [1, n] \\\\ & \sum_i k_i - 1 = 0 \\\\ & \lambda_i \ge 0, \beta_i \ge 0 &\forall i \in [1, n] \\\\ & \lambda_i (r(s, v_i) / k_i - b) = 0 &\forall i \in [1, n] \\\\ &\beta_i (0 - k_i) = 0 &\forall i \in [1, n] \\\\ & \frac{\partial L}{\partial b} = 1 - \sum_{i = 1}^n \lambda_i = 0 \\\\ & \frac{\partial L}{\partial k_i} = -\lambda_i r(s, v_i) / k_i^2 - \beta_i + \mu = 0 &\forall i \in [1, n]\\\\ \end{align}
